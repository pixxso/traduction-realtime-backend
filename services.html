document.getElementById('booking-form').addEventListener('submit', e => { e.preventDefault(); const date = document.getElementById('date').value; const time = document.getElementById('time').value; const specialty = document.getElementById('specialty').value; fetch('/api/reservations', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ date, time, specialty }) }) .then(res => res.text()) .then(msg => { document.getElementById('confirmation-message').textContent = msg; }) .catch(() => { document.getElementById('confirmation-message').textContent = "Erreur lors de la réservation."; }); });
